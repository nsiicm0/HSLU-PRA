# convert text to morse code
def morse_encode(text):
    return ' '.join([''.join(morse_dict[i]) for i in text.lower()])
 
# convert morse code to text
def morse_decode(code):
    #return ''.join([ascii_dict[i] for i in code.split(' ')])
    return ''.join([ascii_dict.get(i,'?') for i in code.split(' ')])

def read_audio(path='morse.wav'):
    fs, data = wavfile.read(path)
    return data,fs

def plot_spectrogram(f,t,S):
    plt.figure(0) # spectrogram
    plt.pcolormesh(t, f, S)
    plt.ylabel('frequency [Hz]')
    plt.xlabel('time [sec]')
    plt.title('spectrogram')
    plt.show()
    
def get_spectrogram(data,fs,plot=True):
    # spectral representation
    f, t, S = signal.spectrogram(data, fs)
    if plot:
        plot_spectrogram(f,t,S)
    return f, t, S

# generate data for training 

def generate_data(n):
    output_list = [(''.join([random.choice(string.ascii_lowercase + '-') 
        for _ in range(word_len)])) 
            for _ in range(n)]
    return  output_list, [morse_encode(s) for s in output_list]
    
# embedding, simply my mapping a one-hot encoding
class Embedding(object):
    def __init__(self, chars):
        self.chars = sorted(set(chars))
        self.char_indices = dict((c, i) for i, c in
                                            enumerate(self.chars))
        self.indices_char = dict((i, c) for i, c in 
                                            enumerate(self.chars))
    def encode(self, token, num_rows):
        x = np.zeros((num_rows, len(self.chars)))
        for i, c in enumerate(token):
            x[i, self.char_indices[c]] = 1
        return x

    def decode(self, x, calc_argmax=True):
        if calc_argmax:
            x = [x.argmax(axis=-1)]
        return ''.join(self.indices_char[int(v)] for v in x)

# -----------------------------------------------------------------------------
# generate & prepare data for feeding to network, i.e. represent by one-hot encoding

def prepare_data(n=10000, in_chars='-. ', out_chars = ''.join(alphabet)):
    # character sets used at input (morse code) and output (text)
    chars_in = in_chars
    chars_out = out_chars
    embedding_in = Embedding(chars_in)
    embedding_out = Embedding(chars_out)

    output_list, input_list = generate_data(n)
    # x : input to encoder, y : output from decoder
    x = np.zeros((len(input_list), max_len_x, len(chars_in)))
    y = np.zeros((len(output_list), max_len_y, len(chars_out)))

    for i, token in enumerate(input_list):
        x[i] = embedding_in.encode(token, max_len_x)
        
    for i, token in enumerate(output_list):
        y[i] = embedding_out.encode(token, max_len_y)
        
    # split data set : 3/4 for training, 1/4 for evaluation
    m = 3*len(x)// 4
    (x_train, x_val) = x[:m], x[m:]
    (y_train, y_val) = y[:m], y[m:]

    # show some examples ...
    for i in range(5):
        print(''.join([embedding_in.decode(token) for token in x_train[i]]),
              ''.join([embedding_out.decode(token) for token in y_train[i]]))
        
    data = dict()
    data['x_train'] = x_train
    data['y_train'] = y_train
    data['x_val'] = x_val
    data['y_val'] = y_val
    data['embedding_in'] = embedding_in
    data['embedding_out'] = embedding_out
    data['chars_in'] = chars_in
    data['chars_out'] = chars_out
    
    return data

# -----------------------------------------------------------------------------
# train model  

def train(model, data, epochs = 80, batch = 50):
    Epochs = epochs
    Batch_size = batch

    x_train = data['x_train']
    y_train = data['y_train']
    x_val = data['x_val']
    y_val = data['y_val']

    # train
    hist = model.fit(x_train, y_train, 
                    batch_size=Batch_size, epochs=Epochs, 
                    validation_data=(x_val, y_val))
    return hist

#show history of progress
def show_progress(hist):
    plt.figure(figsize=(15,5))
    plt.subplot(121)
    plt.plot(hist.history['accuracy'])
    plt.plot(hist.history['val_accuracy'])
    plt.title('model accuracy')
    plt.ylabel('accuracy')
    plt.xlabel('epoch')
    plt.legend(['train', 'validation'], loc='upper left')
    plt.subplot(122)
    plt.plot(hist.history['loss'])
    plt.plot(hist.history['val_loss'])
    plt.title('model loss')
    plt.ylabel('loss')
    plt.xlabel('epoch')
    plt.legend(['train', 'validation'], loc='upper right')
    plt.show()

It seems that changing the latent_dim parameter in the model configuration from 200 to 300 had a positive effect on the models performance. For testing purposes (not shown here), I also tried setting this value to 100, which did not increase the model performance.

Since the model training is quite slow on CPUs, I tried to play around with the batch size for training (large batch should speed up training progress but might slow down convergence). The model with the large batch training seems to actually have been slower in training progress and also has converged slower. So changing this parameter did not improve convergence time. Actually, the larger batches made the model converge on a validation accuracy of around 0.9964 (+/- 0.0002), whereas the previous model with batch size 50 seem to have converged around a validation accuracy of 0.9991 (+/- 0.0001).
